## With batch actions

<Playground>
  <Table
    {...{
      data,
      columns,
    }}
    selectable
  >
    <Table.Toolbar>
      {toolbarProps => {
        const handleAddNew = () => {
          console.log(`add new row`);
        };

        return (
          <Button
            variant="ghost"
            iconPosition="right"
            onClick={handleAddNew}
          >
            Add new <Icon Asset={SvgPlus} />
          </Button>
        );
      }}
    </Table.Toolbar>
    <Table.BatchActionsToolbar>
      {(batchActionProps) => {
        const handleDeleteRows = () => {
          console.log(`delete ${batchActionProps.selectedRows.length}`);
        };

        const handleEditRows = () => {
          console.log(`edit ${batchActionProps.selectedRows.length}`);
        };

        return (
          <Fragment>
            <Button variant="ghost" iconPosition="right" onClick={handleDeleteRows}>
              Delete <Icon Asset={SvgDelete} />
            </Button>
            <Button variant="ghost" iconPosition="right" onClick={handleEditRows}>
              Edit <Icon Asset={SvgDelete} />
            </Button>
          </Fragment>
        );
      }}
    </Table.BatchActionsToolbar>
    <Table.Rows />
    <Table.Pagination />

  </Table>
</Playground>
