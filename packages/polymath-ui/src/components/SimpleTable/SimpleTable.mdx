  {() => {
    const {
      TableContainer,
      Table,
      TableHead, 
      TableHeader, 
      TableBody, 
      TableRow, 
      TableCell
    } = SimpleTable;
    
    const headersData = [
      {
        key: 'tierStatus',
        header: '',
      },
      {
        key: 'tier',
        header: '# Tier',
      },
      {
        key: 'tokenPrice',
        header: <Box textAlign="right">Token Price</Box>,
      },
      {
        key: 'raiseTarget',
        header: <Box textAlign="right">Total Raise Target</Box>,
      },
      {
        key: 'raised',
        header: <Box textAlign="right">Raised</Box>,
      },
      {
        key: 'progress',
        header: ''
      },
    ];

    const rowsData = [
      {
        id: 'a',
        rowProps: {
          status: 'done'
        },
        tier: 1,
        tierStatus: (
          <Box mt="-1px">
            <TierStatus status={'done'} width="16px" height="16px" />
          </Box>
        ),
        tokenPrice: <Box textAlign="right">100</Box>,
        raiseTarget: <Box textAlign="right">300000</Box>,
        raised: <Box textAlign="right">100</Box>,
        progress: (
          <InlineFlex>
            <Box width="150px" mr={1}>
              <ProgressBar height="10px" progress={0.5} />
            </Box>
            100%
          </InlineFlex>
        )
      },
      {
        id: 'b',
        rowProps: {
          status: 'active'
        },
        tierStatus: (
          <Box mt="-1px">
            <TierStatus status={'active'} width="16px" height="16px" />
          </Box>
        ),
        tier: 2,
        tokenPrice: <Box textAlign="right">100</Box>,
        raiseTarget: <Box textAlign="right">300000</Box>,

        progress: (
          <InlineFlex>

              <ProgressBar height="10px" progress={0.5} />

            100%
          </InlineFlex>
        )
      },
      {
        id: 'c',
        rowProps: {
          status: 'not-started'
        },
        tierStatus: (
          <Box mt="-1px">
            <TierStatus status={'not-started'} width="16px" height="16px" />
          </Box>
        ),
        tier: 3,
        tokenPrice: <Box textAlign="right">100</Box>,
        raiseTarget: <Box textAlign="right">300000</Box>,
        raised: <Box textAlign="right">100</Box>,

          <InlineFlex>
            <Box width="150px" mr={1}>
              <ProgressBar height="10px" progress={0.5} />
            </Box>
            100%
          </InlineFlex>
        )
      }
    ];

    return (
      <SimpleTable
        rows={rowsData}
        headers={headersData}
        render={({ rows, headers }) => (
          <Table>
            <TableHead>
              <TableRow>
                {headers.map(header => (
                  <TableHeader>
                    {header.header}
                  </TableHeader>
                ))}
              </TableRow>
            </TableHead>
            <TableBody>
              {rows.map(row => {
                const rowData = rowsData.find(rowData => rowData.id === row.id);
                
                return (
                  <TableRow key={row.id} isActive={rowData.rowProps.status === 'active'}>
                    {row.cells.map(cell => (
                      <TableCell key={cell.id}>{cell.value}</TableCell>
                    ))}
                  </TableRow>
                );
              })}
            </TableBody>
          </Table>
        )}
      />
    );
  }}
